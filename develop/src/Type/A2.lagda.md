## A particular implementation (for HoTT)

```agda
data A2Alphabet : Set where
  ΠF
   ΠI
   ΠE

   ΣF
   ΣI
   ΣE

   +F
   +Iˡ
   +Iʳ
   +E

   𝟘F
   𝟘E

   𝟙F
   𝟙I
   𝟙E

   ℕF
   ℕIᶻ
   ℕIˢ
   ℕE

   =F
   =I
   =E
   : A2Alphabet

interpretAlphabet : A2Alphabet → ∃ (Vec Nat)

interpretAlphabet ΠF  = (_ ,, (0 ∷ 1 ∷ [])             )
interpretAlphabet ΠI  = (_ ,, (0 ∷ 1 ∷ [])             )
interpretAlphabet ΠE  = (_ ,, (0 ∷ 0 ∷ [])             )

interpretAlphabet ΣF  = (_ ,, (0 ∷ 1 ∷ [])             )
interpretAlphabet ΣI  = (_ ,, (0 ∷ 0 ∷ [])             )
interpretAlphabet ΣE  = (_ ,, (1 ∷ 2 ∷ 0 ∷ [])         )

interpretAlphabet +F  = (_ ,, (0 ∷ 0 ∷ [])             )
interpretAlphabet +Iˡ = (_ ,, (0 ∷ [])                 )
interpretAlphabet +Iʳ = (_ ,, (0 ∷ [])                 )
interpretAlphabet +E  = (_ ,, (1 ∷ 1 ∷ 1 ∷ 0 ∷ [])     )

interpretAlphabet 𝟘F  = (_ ,, []                       )
interpretAlphabet 𝟘E  = (_ ,, (1 ∷ 0 ∷ [])             )

interpretAlphabet 𝟙F  = (_ ,, []                       )
interpretAlphabet 𝟙I  = (_ ,, []                       )
interpretAlphabet 𝟙E  = (_ ,, (1 ∷ 0 ∷ 0 ∷ [])         )

interpretAlphabet ℕF  = (_ ,, []                       )
interpretAlphabet ℕIᶻ = (_ ,, []                       )
interpretAlphabet ℕIˢ = (_ ,, (0 ∷ [])                 )
interpretAlphabet ℕE  = (_ ,, (1 ∷ 0 ∷ 2 ∷ 0 ∷ [])     )

interpretAlphabet =F  = (_ ,, (0 ∷ 0 ∷ 0 ∷ [])         )
interpretAlphabet =I  = (_ ,, (0 ∷ [])                 )
interpretAlphabet =E  = (_ ,, (3 ∷ 1 ∷ 0 ∷ 0 ∷ 0 ∷ []) )
```
